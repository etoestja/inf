\begin{frame}{Выборка}
\begin{enumerate}
\item Вероятностные распределения $\{f_k(x)\colon\mathbb{R}\to\mathbb{R}\}_{k=1}^K$
\item Выбирается $k\in\overline{1,K}$
\item Выбираются $[a,b]\subseteq \mathbb{R}$
\item $[a,b]$ разбивается на 101 часть: $[a,b]=\bigsqcup\limits_{s=0}^{100}\Delta_s$, $$|\Delta_0|=|\Delta_{100}|=\frac{\Delta_s}{2}$$
\item Генерируются $t_z\sim f_k$
\item Вычисляется количество объектов в $\Delta_s$: $x^i_s=\sum\limits_z[t_z\in\Delta_s]$

\end{enumerate}
\centering $\Rightarrow$ Получен объект выборки $\{x_s\}\in\mathfrak{D}$
\end{frame}
