#include <iostream>
using namespace std;

int n, k, s;

bool Possible (int i, int j)
{
	if (n==2)
		return (s==i+j);
	int k1 = (s-i-j)/(n-2);
	int k2 = 0;
	if ((s-i-j)%(n-2) == 0)
		k2 = k1;
	else
		k2 = k1+1;
	return ((s-i-j) <= (n-2)*k) && (i <= k1) && (j >= k1) && (i <= k2) && (j >= k2);
}

int main()
{
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	cin >> n >> k >> s;
	if (n == 1)
	{
		cout << s << ' ' << s << ' ' << s << ' ' << s << endl;
		return 0;
	}
	int a1=k+1, a2=0, a3=k+1, a4=0;
	for (int i = 1; i <= k; i++)
		for (int j = k; j >= i; j--)
			if (Possible(i, j))
			{
				//cout << i << ' ' << j << endl;
				a1 = min(a1, i);
				a2 = max(a2, i);
				a3 = min(a3, j);
				a4 = max(a4, j);
			}
	cout << endl;
	cout << a1 << ' ' << a2 << ' ' << a3 << ' ' << a4 << endl;
	return 0;
}
